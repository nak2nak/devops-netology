 Создан init-файл для автозагрузки

    [Unit]
    Description=node_exporter
    Wants=network-online.target
    After=network-online.target 
    
    [Service]
    Type=simple
    User=nod_exporter
    Group=nod_exporter
    ExecStart=/usr/local/bin/node_exporter 
    Restart=always
    RestartSec=10s
    NotifyAccess=all 
    
    [Install]
    WantedBy=multi-user.target
